name: CI - UI Build

on:
  push:
    branches:
      - main
      - recovery/render-worker-bridge-pass
    paths:
      - 'ui/**'
      - '.github/workflows/ci-ui-build.yml'
  pull_request:
    branches:
      - main
    paths:
      - 'ui/**'
      - '.github/workflows/ci-ui-build.yml'
  workflow_dispatch:

permissions:
  contents: read

concurrency:
  group: ci-ui-build-${{ github.ref }}
  cancel-in-progress: true

jobs:
  build-ui:
    name: UI build syntax-safe
    runs-on: ubuntu-latest
    timeout-minutes: 20
    env:
      FORCE_JAVASCRIPT_ACTIONS_TO_NODE24: 'true'
    defaults:
      run:
        working-directory: ui
        shell: bash
    steps:
      - name: Check out repository
        uses: actions/checkout@v5

      - name: Set up Node.js
        uses: actions/setup-node@v6
        with:
          node-version: '24'
          cache: 'npm'
          cache-dependency-path: ui/package-lock.json

      - name: Show runtime
        run: |
          node --version
          npm --version
          pwd
          ls -la

      - name: Install UI dependencies
        run: npm ci --include=dev

      - name: Verify Vite exists
        run: ls -la node_modules/vite/bin

      - name: Typecheck audit lane
        run: npm run typecheck

      - name: Build UI
        run: npm run build

      - name: Verify dist artifact
        run: |
          test -f dist/index.html
          find dist -maxdepth 2 -type f | head -50
