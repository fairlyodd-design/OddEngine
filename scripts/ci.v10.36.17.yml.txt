name: OddEngine CI Recovery

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]
  workflow_dispatch:

permissions:
  contents: read

concurrency:
  group: oddengine-ci-${{ github.ref }}
  cancel-in-progress: true

# v10.36.17 — opt into Node 24 action runtime now so the workflow is ready
# for the 2026 GitHub Actions Node runtime migration.
env:
  FORCE_JAVASCRIPT_ACTIONS_TO_NODE24: 'true'
  NPM_CONFIG_AUDIT: 'false'
  NPM_CONFIG_FUND: 'false'

jobs:
  runtime-build:
    name: Runtime audit + UI build
    runs-on: ubuntu-latest
    timeout-minutes: 20

    steps:
      - name: Checkout
        uses: actions/checkout@v6

      - name: Setup Node
        uses: actions/setup-node@v6
        with:
          node-version: '24'
          cache: npm
          cache-dependency-path: |
            ui/package-lock.json

      - name: Show runtime
        run: |
          set -euxo pipefail
          node --version
          npm --version
          pwd
          ls -la
          echo "GITHUB_SHA=$GITHUB_SHA"
          echo "GITHUB_REF=$GITHUB_REF"

      - name: Validate OddEngine shape
        run: |
          set -euxo pipefail
          test -f package.json
          test -d ui
          test -f ui/package.json
          test -f ui/package-lock.json
          test -f ui/tsconfig.audit.json
          test -f ui/src/auditEntry.ts
          test -f scripts/system-runtime-import-audit.mjs

      - name: Install UI dependencies
        run: |
          set -euxo pipefail
          npm --prefix ui install --no-audit --no-fund

      - name: Runtime import audit
        run: |
          set -euxo pipefail
          node scripts/system-runtime-import-audit.mjs

      - name: Typecheck audit lane
        run: |
          set -euxo pipefail
          npm --prefix ui run typecheck

      - name: Build UI
        run: |
          set -euxo pipefail
          npm --prefix ui run build

      - name: Verify dist artifact
        run: |
          set -euxo pipefail
          test -f ui/dist/index.html
          find ui/dist -maxdepth 2 -type f | head -50

      - name: Upload UI dist
        uses: actions/upload-artifact@v6
        with:
          name: oddengine-ui-dist
          path: ui/dist
          if-no-files-found: error

      - name: Failure breadcrumbs
        if: failure()
        run: |
          echo "OddEngine CI failed before completion. Check the first failing step above this one."
          echo "Node: $(node --version || true)"
          echo "NPM: $(npm --version || true)"
          echo "Repo root: $(pwd)"
          echo "Git status:"
          git status --short || true
          echo "UI package summary:"
          node -e "const p=require('./ui/package.json'); console.log({name:p.name, version:p.version, scripts:p.scripts})" || true
