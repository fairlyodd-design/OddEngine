name: CI - UI Build

on:
  push:
    branches:
      - main
      - recovery/render-worker-bridge-pass
    paths:
      - 'ui/**'
      - '.github/workflows/ci-ui-build.yml'
  pull_request:
    branches:
      - main
    paths:
      - 'ui/**'
      - '.github/workflows/ci-ui-build.yml'
  workflow_dispatch:

permissions:
  contents: read

concurrency:
  group: ci-ui-build-${{ github.ref }}
  cancel-in-progress: true

jobs:
  build-ui:
    name: UI build install-safe
    runs-on: ubuntu-latest
    timeout-minutes: 20
    env:
      FORCE_JAVASCRIPT_ACTIONS_TO_NODE24: 'true'
    defaults:
      run:
        working-directory: ui
        shell: bash
    steps:
      - name: Check out repository
        uses: actions/checkout@v5

      - name: Set up Node.js
        uses: actions/setup-node@v6
        with:
          node-version: '24'
          cache: 'npm'
          cache-dependency-path: ui/package-lock.json

      - name: Show runtime
        run: |
          set -euxo pipefail
          node --version
          npm --version
          pwd
          ls -la
          echo "package.json version:"
          node -e "const p=require('./package.json'); console.log(p.name, p.version || 'no-version')"

      - name: Install UI dependencies
        run: |
          set -euxo pipefail
          npm install

      - name: Verify Vite exists
        run: |
          set -euxo pipefail
          test -f node_modules/vite/bin/vite.js
          ls -la node_modules/vite/bin

      - name: Typecheck audit lane
        run: |
          set -euxo pipefail
          npm run typecheck

      - name: Build UI
        run: |
          set -euxo pipefail
          npm run build

      - name: Verify dist artifact
        run: |
          set -euxo pipefail
          test -f dist/index.html
          find dist -maxdepth 2 -type f | head -50

      - name: Failure breadcrumbs
        if: failure()
        run: |
          echo "CI UI build failed. Breadcrumbs:"
          node --version || true
          npm --version || true
          pwd || true
          ls -la || true
          test -f package.json && cat package.json || true
          test -f package-lock.json && node -e "const l=require('./package-lock.json'); console.log('lockfileVersion', l.lockfileVersion);" || true
